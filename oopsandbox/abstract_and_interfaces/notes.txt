Tea class before reconfigure
In the above two classes the three methods addHotWater(), addSugar(), and addMilk() are same. So we should remove duplicated code.
class Tea
{
    public function addTea()
    {
        var_dump('Add proper amount of tea');
        return $this;
    }
    protected  function  addHotWater()
    {
        var_dump('Pour Hot water into cup');
        return $this;
    }

    protected  function addSugar()
    {
        var_dump('Add proper amount of sugar');
        return $this;
    }

    protected function addMilk()
    {
        var_dump('Add proper amount of Milk');
        return $this;
    }

    public function make()
    {
        return $this
            ->addHotWater()
            ->addSugar()
            ->addTea()
            ->addMilk();
    }
}
